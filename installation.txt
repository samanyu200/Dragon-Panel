# 1️⃣ Clone the Dragon Panel repository
git clone https://github.com/samanyu200/Dragon-Panel.git

# 2️⃣ Move into the panel directory
cd Dragon-Panel

# 3️⃣ Install dependencies
npm install

# 4️⃣ Seed the database (if needed)
npm run seed

# 5️⃣ Create an admin user
npm run createUser

# 6️⃣ Start the panel on port 3000
node .
